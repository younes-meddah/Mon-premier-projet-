function trierRecettesParNom(recettes) {
  return recettes.sort((a, b) => {
    A = a.name.toUpperCase(); // Convertir les noms en majuscules
    B = b.name.toUpperCase(); // Convertir les noms en majuscules

    if (A < B) {
      return -1;
    }
    if (A > B) {
      return 1;
    }
    return 0; // Les noms sont égaux
  });
}

function moyenne(recipe){
	j=0;
	rating=0;
	for(j=0;j<recipe.comments.length;j++){
	rating=rating+recipe.comments[j].rating;}
	rating=rating/recipe.comments.length;
	const rating1=Math.floor(rating);
	return rating1;
}

function rechercherRecettes(name, categorie ,pays ,duree ,rating) {
  return recettesDB.filter(function(recette) {
	  urlParams = new URLSearchParams(window.location.search);
	  if(urlParams.get('rating')!=''){
    return recette.name.toLowerCase().includes(name.toLowerCase()) &&
           recette.category.toLowerCase().includes(categorie.toLowerCase()) &&
		   recette.country.toLowerCase().includes(pays.toLowerCase()) && 
		   recette.duration.toLowerCase().includes(duree.toLowerCase()) &&
	  comparer(parseFloat(rating),moyenne(recette));}
	  else{
		  return recette.name.toLowerCase().includes(name.toLowerCase()) &&
           recette.category.toLowerCase().includes(categorie.toLowerCase()) &&
		   recette.country.toLowerCase().includes(pays.toLowerCase()) && 
		   recette.duration.toLowerCase().includes(duree.toLowerCase())
	  }
  });
}
