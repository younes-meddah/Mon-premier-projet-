const randomRecipes = () => {
  let randomRecipe1 = Math.floor(Math.random() * recipes.length);
  let randomRecipe2 = Math.floor(Math.random() * recipes.length);
  let randomRecipe3 = Math.floor(Math.random() * recipes.length);

  while (randomRecipe1 === randomRecipe2) {
    randomRecipe2 = Math.floor(Math.random() * recipes.length);
  }

  while (randomRecipe1 === randomRecipe3 || randomRecipe2 === randomRecipe3) {
    randomRecipe3 = Math.floor(Math.random() * recipes.length);
  }
  document.getElementById("recipe1").href='detail.html?id='+((randomRecipe1+1)%12);
  document.getElementById("recipe1-title").innerHTML = recipes[randomRecipe1].title;
  document.getElementById("recipe1-image").src = recipes[randomRecipe1].image;
  document.getElementById("recipe1-description").innerHTML = recipes[randomRecipe1].description;
  document.getElementById("recipe2").href='detail.html?id='+((randomRecipe2+1)%12);
  document.getElementById("recipe2-title").innerHTML = recipes[randomRecipe2].title;
  document.getElementById("recipe2-image").src = recipes[randomRecipe2].image;
  document.getElementById("recipe2-description").innerHTML = recipes[randomRecipe2].description;
  document.getElementById("recipe3").href='detail.html?id='+((randomRecipe3+1)%12);
  document.getElementById("recipe3-title").innerHTML = recipes[randomRecipe3].title;
  document.getElementById("recipe3-image").src = recipes[randomRecipe3].image;
  document.getElementById("recipe3-description").innerHTML = recipes[randomRecipe3].description;
}
